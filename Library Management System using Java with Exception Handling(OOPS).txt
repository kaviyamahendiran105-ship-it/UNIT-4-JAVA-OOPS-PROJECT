import java.util.Scanner;

// Custom Exceptions
class BookNotFoundException extends Exception {
    BookNotFoundException(String msg) {
        super(msg);
    }
}

class BookAlreadyIssuedException extends Exception {
    BookAlreadyIssuedException(String msg) {
        super(msg);
    }
}

class BookNotIssuedException extends Exception {
    BookNotIssuedException(String msg) {
        super(msg);
    }
}

// Abstract class
abstract class LibraryOperations {
    abstract void issueBook(int id) throws BookNotFoundException, BookAlreadyIssuedException;
    abstract void returnBook(int id) throws BookNotFoundException, BookNotIssuedException;
}

// Book Class (Encapsulation)
class Book {
    private int id;
    private String name;
    private String author;
    private boolean issued;

    Book(int id, String name, String author) {
        this.id = id;
        this.name = name;
        this.author = author;
        this.issued = false;
    }

    public int getId() {
        return id;
    }

    public String getName() {
        return name;
    }

    public String getAuthor() {
        return author;
    }

    public boolean isIssued() {
        return issued;
    }

    public void setIssued(boolean status) {
        issued = status;
    }
}

// Library Class
class Library extends LibraryOperations {

    Book books[] = new Book[10];
    int count = 0;

    void addBook(int id, String name, String author) {
        books[count] = new Book(id, name, author);
        count++;
        System.out.println("Book added successfully");
    }

    // Issue Book
    void issueBook(int id) throws BookNotFoundException, BookAlreadyIssuedException {
        for (int i = 0; i < count; i++) {

            if (books[i].getId() == id) {

                if (books[i].isIssued()) {
                    throw new BookAlreadyIssuedException("Book already issued");
                }

                books[i].setIssued(true);
                System.out.println("Book issued successfully");
                return;
            }
        }
        throw new BookNotFoundException("Book not found");
    }

    // Return Book
    void returnBook(int id) throws BookNotFoundException, BookNotIssuedException {
        for (int i = 0; i < count; i++) {

            if (books[i].getId() == id) {

                if (!books[i].isIssued()) {
                    throw new BookNotIssuedException("Book was not issued");
                }

                books[i].setIssued(false);
                System.out.println("Book returned successfully");
                return;
            }
        }
        throw new BookNotFoundException("Book not found");
    }

    // Display Books
    void displayBooks() {
        for (int i = 0; i < count; i++) {

            System.out.println("Book ID: " + books[i].getId());
            System.out.println("Book Name: " + books[i].getName());
            System.out.println("Author: " + books[i].getAuthor());
            System.out.println("Issued: " + books[i].isIssued());
            System.out.println();
        }
    }
}

// Main Class
public class Main {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        Library lib = new Library();
        int choice;

        do {
            System.out.println("\nLibrary Management System");
            System.out.println("1 Add Book");
            System.out.println("2 Issue Book");
            System.out.println("3 Return Book");
            System.out.println("4 Display Books");
            System.out.println("5 Exit");

            System.out.println("Enter your choice:");
            choice = sc.nextInt();

            try {

                if (choice == 1) {
                    System.out.println("Enter Book ID:");
                    int id = sc.nextInt();
                    sc.nextLine();

                    System.out.println("Enter Book Name:");
                    String name = sc.nextLine();

                    System.out.println("Enter Author:");
                    String author = sc.nextLine();

                    lib.addBook(id, name, author);
                }

                else if (choice == 2) {
                    System.out.println("Enter Book ID to Issue:");
                    int id = sc.nextInt();
                    lib.issueBook(id);
                }

                else if (choice == 3) {
                    System.out.println("Enter Book ID to Return:");
                    int id = sc.nextInt();
                    lib.returnBook(id);
                }

                else if (choice == 4) {
                    lib.displayBooks();
                }

            } catch (BookNotFoundException e) {
                System.out.println(e.getMessage());
            } catch (BookAlreadyIssuedException e) {
                System.out.println(e.getMessage());
            } catch (BookNotIssuedException e) {
                System.out.println(e.getMessage());
            } finally {
                System.out.println("Operation Completed");
            }

        } while (choice != 5);

        System.out.println("Program Ended");
    }
}
